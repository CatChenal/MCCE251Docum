210  to setup, call the setup functions before get into the vdw loop. See example:
211  */
212  float pair_vdw = 0., d2, d6, d12, e, C6, C12;
213  VECTOR v1,v2;
214  int i_atom, j_atom, i_connect, done;
215  ATOM *iatom_p,*jatom_p;
216  float cutoff_near2 = VDW_CUTOFF_NEAR * VDW_CUTOFF_NEAR;
217  float cutoff_far2 = VDW_CUTOFF_FAR * VDW_CUTOFF_FAR;
218

220  if (!prot.res[j_res].conf[j_conf].n_atom) return pair_vdw;

222  if (!prot.res[j_res].cal_vdw) return pair_vdw;
223 
224  for (i_atom=0; i_atom<prot.res[i_res].conf[i_conf].n_atom; i_atom++) {
225  iatom_p = &prot.res[i_res].conf[i_conf].atom[i_atom];
226  if (!iatom_p->on) continue;
227  v1 = iatom_p->xyz;
228  for (j_atom=0; j_atom<prot.res[j_res].conf[j_conf].n_atom; j_atom++) {
229  jatom_p = &prot.res[j_res].conf[j_conf].atom[j_atom];
230  if (!jatom_p->on) continue;
231  v2 = jatom_p->xyz;
232 
233  if (iatom_p == jatom_p) continue;
234  d2 = ddvv(v1, v2);
235  if (d2 < cutoff_far2) {
236  done = 0;
237 
238  if (d2 <= prot.res[i_res].r12sq_max) {
239  for (i_connect=0;i_connect<prot.res[i_res].n_connect12[i_conf][i_atom];i_connect++) {
240  if (jatom_p == prot.res[i_res].connect12[i_conf][i_atom][i_connect]) {
241  done = 1;
242  }
243  }
244  }
245  if (done) continue;

247  if (d2 <= prot.res[i_res].r13sq_max) {
248  for (i_connect=0;i_connect<prot.res[i_res].n_connect13[i_conf][i_atom];i_connect++) {
249  if (jatom_p == prot.res[i_res].connect13[i_conf][i_atom][i_connect]) {
250  done = 1;
251  }
252  }
253  }
254  if (done) continue;
255 
256  /* calculate vdw */
257  if (d2 < cutoff_near2)
258  e = VDW_ELIMIT_NEAR; /* Cutoff */
259  else {
260  //C6 = C6_matrix[iatom_p->i_elem][jatom_p->i_elem];
261  //C12 = C12_matrix[iatom_p->i_elem][jatom_p->i_elem];
262  float sig_min = atom1.vdw_rad + atom2.vdw_rad;
263  float eps = sqrt(atom1.vdw_eps*atom2.vdw_eps);
264  C12 = eps*pow(sig_min,12);
265  C6 = 2.*eps*pow(sig_min,6);
266

268  d12 = d6*d6;
269  e = C12/d12 - C6/d6;
270  }

273  for (i_connect=0;i_connect<prot.res[i_res].n_connect14[i_conf][i_atom];i_connect++) {
274  if (jatom_p == prot.res[i_res].connect14[i_conf][i_atom][i_connect]) {
275 
276  pair_vdw += e * env.factor_14lj;
277  done = 1;
278  break;
279  }
280  }
281  }
282  if (done) continue;
283 
284  pair_vdw += e;
285  }
286  }
287  }
