3  * db_open - open mcce parameter database

5  * SYNOPSIS
6  * #include <gdbm.h>
7  * #include <mcce.h>

9  * int db_open();
10  *
11  * DESCRIPTION
12  * The db_open() function opens a paramter database for fast access to the
13  * parameter entries with the support of standard gdbm library. If this
14  * function is called while the database is already open, it prints out
15  * a warning message and use the existing database. Use "-lgdbm" option
16  * to compile these programs, just like use "-lm" to compile programs
17  * calling math functions.
18  *
19  * RETURN VALUE
20  * The returned integer is the status, 0 is success, other numbers are the
21  * error code defined by gdbm.h.
22  *
23  * SEE ALSO
24  * db_close, iatom, param_get, param_sav
25  *
26  * EXAMPLE
27  * #include <stdio.h>
28  * #include <gdbm.h>
29  * #include "mcce.h"
30  *
31  * int main()
32  * { char conformers[] = "ASP0, ASP1, ASP-";
33  * int i = 10;
34  * float pKa = 5.7;
35  *
36  * char ret_conf[160];
37  * int ret_i = 0;
38  * float ret_pKa = 0.0;
39  *
40  * db_open();
41  *
42  * param_sav("CONFLIST", "ASP", "", &conformers, sizeof(conformers));
43  * param_sav("IATOM", "ASP0", "CG", &i, sizeof(int));
44  * param_sav("PKA", "ASP-", "", &pKa, sizeof(float));
45  *
46  * param_get("CONFLIST", "ASP", "", &ret_conf);

48  *

54  *
