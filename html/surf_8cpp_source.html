339 int mkacc(ATOM *atom)
340 {
341  VECTOR point;
342  int i, j, k, l, m;
343  float d_square;
344  int count;
345  int ix, iy, iz;
346  int ax, ay, az; /* the grid index of this atom */
347  int status;
348 
349  count = num_pts;
350 
351  ax = (int) ((atom -> xyz.x - xyz_min.x) / grid_interval);
352  ay = (int) ((atom -> xyz.y - xyz_min.y) / grid_interval);
353  az = (int) ((atom -> xyz.z - xyz_min.z) / grid_interval);
354 
355  /* loop over each point */
356 
357  for (m = 0; m < num_pts; m++) {
358  status = TRUE;
359 
360  point.x = point_preset[m][0] * atom -> rad + atom -> xyz.x;
361  point.y = point_preset[m][1] * atom -> rad + atom -> xyz.y;
362  point.z = point_preset[m][2] * atom -> rad + atom -> xyz.z;
363 
364  /* find out which grid this point belongs to */
365  ix = (int) ((point.x - xyz_min.x) / grid_interval);
366  iy = (int) ((point.y - xyz_min.y) / grid_interval);
367  iz = (int) ((point.z - xyz_min.z) / grid_interval);
368 
369  if (grid[ix][iy][iz].n_atom >= 1) {
370  for (l = 0; l < grid[ix][iy][iz].n_atom; l++) {
371  if (grid[ix][iy][iz].atom[l].on &&
372  status == TRUE &&
373  (ix != ax || iy != ay || iz != az ||
374  memcmp(atom, &(grid[ix][iy][iz].atom[l]), sizeof(ATOM)))) {
375  d_square = ddvv(point, grid[ix][iy][iz].atom[l].xyz);
376  if (d_square < grid[ix][iy][iz].atom[l].rad * grid[ix][iy][iz].atom[l].rad) {
377  status = FALSE;
378  count --;
379  }
380  }
381  }
382  }
383 
384  for (i = ix - 1; i <= ix + 1; i++) {
385  if (status == FALSE) break;
386  for (j = iy - 1; j <= iy + 1; j++) {
387  if (status == FALSE) break;
388  for (k = iz - 1; k <= iz + 1; k++) {
389  if (status == FALSE) break;
390  for (l = 0; l < grid[i][j][k].n_atom; l++) {
391  if (grid[i][j][k].atom[l].on && status == TRUE) {
392  if (i != ax || j != ay || k != az ||
393  memcmp(atom, &(grid[i][j][k].atom[l]), sizeof(ATOM))) {
394  d_square = ddvv(point, grid[i][j][k].atom[l].xyz);
395  if (d_square < grid[i][j][k].atom[l].rad * grid[i][j][k].atom[l].rad) {
396  status = FALSE;
397  count --;
398  }
399  }
400  }
401  }
402  }
403  }
404  }
405  }
406 
407  atom->sas = area_coeff * atom->rad * atom->rad * (float) count;
408 
409  return 0;
410 }
