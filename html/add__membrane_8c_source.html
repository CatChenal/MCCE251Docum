219 /* move protein so that the anchor is aligned to a defined orientation */
220 int anchor2defined(PROT *prot_p, VECTOR *pos) {
221  float cos_theta, theta;
222 
223  int i_res, i_conf, i_atom, na, ia;
224  ATOM **all_atoms; /* a list of pointers pointing to all atoms in prot */
225  ATOM *prot_anchor;
226 
227  prot_anchor = prot_p->res[prot_p->n_res-1].conf[0].atom;
228  na = 0;
229  all_atoms = NULL;
230  for (i_res=0; i_res<prot_p->n_res; i_res++) {
231  for (i_conf=0; i_conf<prot_p->res[i_res].n_conf; i_conf++) {
232  for (i_atom=0; i_atom<prot_p->res[i_res].conf[i_conf].n_atom; i_atom++) {
233  ATOM *atom_p = &prot_p->res[i_res].conf[i_conf].atom[i_atom];
234  if (!atom_p->on) continue;
235  na++;
236  all_atoms = realloc(all_atoms, na*sizeof(ATOM *));
237  all_atoms[na-1] = atom_p;
238  }
239  }
240  }
241 
242  /* move anchor[0] to pos[0] */
243  VECTOR shift = vector_vminusv(pos[0], prot_anchor[0].xyz);
244  for (ia=0; ia<na; ia++) {
245  all_atoms[ia]->xyz = vector_vplusv(all_atoms[ia]->xyz, shift);
246  }
247 
248  /* rotate anchor[1] to be aligned with pos[1] */
249  /* get i direction given by the defined orientaion (pos) and the current
250  protein position (prot_anchor) */
251  VECTOR vec_axis;
252  VECTOR pos_i, pos_j, pos_k;
253  VECTOR anchor_i, anchor_j;
254 
255  pos_i = vector_normalize(vector_vminusv(pos[1],pos[0]));
256  anchor_i = vector_normalize(vector_vminusv(prot_anchor[1].xyz, prot_anchor[0].xyz));
257 
258  cos_theta = vdotv(anchor_i, pos_i);
259  theta = acos(cos_theta);
260 
261  vec_axis = vector_vplusv(pos[0],vector_vxv(anchor_i, pos_i));
262 
263  rotate_atoms(pos[0], vec_axis, theta, na, all_atoms);
264 
265  /* rotate anchor[2] to be aligned with pos[2] */
266  pos_j = vector_normalize(vector_vminusv(pos[2],pos[0]));
267  anchor_j = vector_normalize(vector_vminusv(prot_anchor[2].xyz, prot_anchor[0].xyz));
268 
269  cos_theta = vdotv(anchor_j, pos_j);
270  theta = acos(cos_theta);
271 
272  pos_k = vector_normalize(vector_vminusv(pos[3],pos[0]));
273  if (vdotv(vector_vxv(anchor_j,pos_j),pos_i) <0.)
274  theta = 2.*env.PI - theta;
275 
276  rotate_atoms(pos[0], pos[1], theta, na, all_atoms);
277 
278  free(all_atoms);
279  na = 0;
280  return 0;
281 }
