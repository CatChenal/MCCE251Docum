476 vector<string> split_w_quote(const string& str)
477 {
478  /* break a line where quote marks can be used to enclose one token especially
479  when spaces occur in tokens */
480  vector<string> tokens; // Create vector to hold the words
481  bool quote_on = false;
482  int last_break = -1;
483 
484  /* when reading character by character, each can be either a space, or a quote mark
485  or a regular character. spaces will break the tokens unless it is enclosed in quote marks */
486  for (int i_char=0; i_char<str.size(); ++i_char) {
487  if (str[i_char] == ' ') {
488  if (quote_on) continue;
489 
490  if ((i_char-last_break) >1) {
491  tokens.push_back(str.substr(last_break+1, i_char-last_break-1));
492  }
493  last_break = i_char;
494  }
495  else if (str[i_char] == '"') {
496  quote_on = not quote_on;
497  if ((i_char-last_break) >1 or not quote_on) {
498  tokens.push_back(str.substr(last_break+1, i_char-last_break-1));
499  }
500  last_break = i_char;
501  }
502  else {
503  continue;
504  }
505  }
506  if (last_break < str.size()-1) {
507  tokens.push_back(str.substr(last_break+1, str.size()-last_break-1));
508  }
509 
510  return tokens;
511 }
