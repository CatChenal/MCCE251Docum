17  * Return integer 0 on success or -1 on failure.

19  * SEE ALSO
20  * load_param, db_open, db_close

22  * EXAMPLE
23  * #include <mcce.h>
24  *
25  * int main()
26  * { db_open();
27  * load_all_param("param_dir");
28  * db_close();
29  * return 0;
30  * }
31  *
32  * DEPENDENCY

43 int load_all_param(char *dirname)
44 { int i;
45  STRINGS files;
46  char fullname[MAXCHAR_LINE];
47 
48  /* Open and read the parameter file names in a directory */
49  files = get_files(dirname);
50  if(files.n == 0) {
51  printf(" FATAL: load_all_param(): \"Invalid or empty directory.\"\n");
52  return USERERR;
53  }
54 
55  /* load each file in this directory */
56  for (i=0; i<files.n; i++) {
57  /* create the full file name with path */
58  //if (!strstr(files.strings[i], ".tpl")) continue;
59  if (strlen(files.strings[i])<=4) continue;
60  if (strcmp(files.strings[i]+(strlen(files.strings[i])-4),".tpl")) continue;
61  sprintf(fullname, "%s/%s", dirname, files.strings[i]);
62 
63  /* load the file */
64  /* printf(" loading \"%s\"\n", fullname);*/
65  if (load_param(fullname)) {
66  printf(" FATAL: load_all_param(): Failed loading file \"%s\".\n", fullname);
67  return USERERR;
68  }
69  }
70 
71  free_strings(&files);
72  return 0;
73 }
