442  for (ic=0; ic<prot.sidechain.size(); ic++) {
443  pairwise[ic].resize(prot.sidechain.size(),0.0);
444  }
445 
446  for (ic=0; ic<prot.sidechain.size(); ic++) {
447  if (prot.sidechain[ic]->uniqID.substr(3,2) == "DM") {
448  continue;
449  }
450 
451  string file_name = ENERGIES;
452  file_name += "/" + prot.sidechain[ic]->uniqID + ".opp";
453  load_opp(pairwise[ic], file_name.c_str(), prot, ic);
454  }
455 
456  printf(" WARNING: Big difference (> %.1f%%) is reported\n", prot.param.monte_warn_pairwise);
457  for (i_res=0; i_res<prot.res.size(); i_res++) {
458  for (i_conf=1; i_conf<prot.res[i_res].conf.size(); i_conf++) {
459  ic = prot.res[i_res].conf[i_conf].i_sidechain_prot;
460 
461  for (j_res=i_res; j_res<prot.res.size(); j_res++) {
462  for (j_conf=1; j_conf<prot.res[j_res].conf.size(); j_conf++) {
463  jc = prot.res[j_res].conf[j_conf].i_sidechain_prot;
464 
465  if (i_res != j_res) {
466  /* warning for asymmetry */
467  float diff = fabs(pairwise[ic][jc]-pairwise[jc][ic]);
468  float avg = (pairwise[ic][jc]+pairwise[jc][ic]);
469  if (diff*100./avg > prot.param.monte_warn_pairwise) {
470  if ( diff > 0.5 ) {
471  // cout << i_res << " " << j_res << endl;
472  printf("%s -> %s = %10.3f, <- = %10.3f\n",
473  prot.sidechain[ic]->uniqID.c_str(),
474  prot.sidechain[jc]->uniqID.c_str(),
475  pairwise[ic][jc], pairwise[jc][ic]);
476  }
477  }
478 
479  if (prot.param.monte_average_pairwise) { /* average */
480  pairwise[ic][jc] = pairwise[jc][ic] = 0.5*(pairwise[ic][jc]+pairwise[jc][ic]);
481  }
482  else { /* get the smaller */
483  pairwise[ic][jc] = pairwise[jc][ic] = fabs(pairwise[ic][jc]) > fabs(pairwise[jc][ic]) ? pairwise[jc][ic] : pairwise[ic][jc];
484  }
485  }
486  else {
487  pairwise[ic][jc] = pairwise[jc][ic] = 0.0; /* Necessary for fast E updating */
488  }
489  }
490  }
491  }
492  }

494  return 0;

655 void Monte_MC(Prot& prot)
656 {
657  /* one cycle of monte carlo runs.
658  All the counters are accumulated.
659  03/02/2007: conf.occ not updated in this subroutine.
660  Although conf.counter_accept is updated here, counter_iter is not kept track of */
661  int k_weighted, k_res, k_conf, k_ngh;
662  int n_flip, i_flip, i_iter;
663  int ic_old, ic_new, jc_w;
664  int j_res;
665  double E_delta;
666 
667  i_iter = prot.param.monte_niter_cycle;
668  while (i_iter) {
669 
670  /* select one residue */
671  k_weighted = (int) (ran2(&dummy_num) * (double)prot.weighted_res_list.size());
672  k_res = prot.weighted_res_list[k_weighted];
673  //k_res = (int) (ran2(&dummy_num) * (double)prot.res.size());
674 
675 
676  /* decide how many flips */
677  n_flip = (int) (ran2(&dummy_num) * (double)prot.res[k_res].n_flip_max) + 1;
678 
679  /* get the list of residue to flip */
680  flip_res[0] = &prot.res[k_res];
681  for (i_flip=1;i_flip<n_flip;i_flip++) {
682  int n_ngh = prot.res[k_res].ngh.size() - (i_flip-1);
683  k_ngh = (int) (ran2(&dummy_num) * (double)n_ngh);
684  flip_res[i_flip] = prot.res[k_res].ngh[k_ngh];
685  swap(prot.res[k_res].ngh[k_ngh], prot.res[k_res].ngh[n_ngh-1]);
686  }
687 
688  /* each residue pick new conformer */
689  for (i_flip=0;i_flip<n_flip;i_flip++) {
690  flip_res[i_flip]->conf_old = flip_res[i_flip]->conf_w;
691 
692  k_conf = (int)(ran2(&dummy_num) * (double)(flip_res[i_flip]->conf.size()-1)) + 1;
693  flip_res[i_flip]->conf_new = &flip_res[i_flip]->conf[k_conf];
694  flip_res[i_flip]->counter_trial++;
695  flip_res[i_flip]->conf_new->counter_trial++;
696  }
697 
698  /* calc. deltaE */
699  E_delta = 0.;
700  for (i_flip=0;i_flip<n_flip;i_flip++) {
701  ic_old = flip_res[i_flip]->conf_old->i_sidechain_prot_orig;
702  ic_new = flip_res[i_flip]->conf_new->i_sidechain_prot_orig;
703 
704  E_delta += (flip_res[i_flip]->conf_new->E_self - flip_res[i_flip]->conf_old->E_self);
705  for (j_res=0; j_res<prot.res.size(); j_res++) {
706  jc_w = prot.res[j_res].conf_w->i_sidechain_prot_orig;
707  E_delta += (pairwise[ic_new][jc_w] - pairwise[ic_old][jc_w]);
708  }
709 
710  /* real flip the active conformer */
711  flip_res[i_flip]->conf_w = flip_res[i_flip]->conf_new;
712  }
713 
714  /* Metropolis */
715  if (exp(-beta*E_delta) > ran2(&dummy_num)) {
716  prot.E_state += E_delta;
717  }
718  else {
719  for (i_flip=0;i_flip<n_flip;i_flip++) {
720  flip_res[i_flip]->conf_w = flip_res[i_flip]->conf_old;
721  }
722  }
723 
724  for (k_res=0;k_res<prot.res.size();k_res++) {
725  prot.res[k_res].conf_w->counter_accept++;
726  }
727  if (prot.E_state < prot.E_min) prot.E_min = prot.E_state;
728  prot.E_accum += prot.E_state;
729 
730  //if (prot.param.monte_do_energy) do_free_energy(prot_p);
731  i_iter--;
732  }
733 }
