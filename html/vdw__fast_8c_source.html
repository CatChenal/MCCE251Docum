413  if (done) continue;
414 
415  /* calculate vdw */
416  if (d2 < cutoff_near2)
417  e = VDW_ELIMIT_NEAR; /* Cutoff */
418  else {
419  float sig_min = iatom_p->vdw_rad + jatom_p->vdw_rad;
420  float eps = sqrt(iatom_p->vdw_eps*jatom_p->vdw_eps);
421  C12 = eps*pow(sig_min,12);
422  C6 = 2.*eps*pow(sig_min,6);
423 
424  d6 = d2*d2*d2;
425  d12 = d6*d6;
426  e = C12/d12 - C6/d6;
427  }
428

432  if (jatom_p == prot.res[i_res].connect14[i_conf][i_atom][i_connect]) {
433 
434  pair_vdw += e * env.factor_14lj;
435 
436  /* print large atom-atom vdw */
437  printf("VDW (1,4) btw \"%s %s\" and \"%s %s\": %8.3f, dist=%8.4f\n",
438  iatom_p->name, prot.res[i_res].conf[i_conf].uniqID,
439  jatom_p->name, prot.res[j_res].conf[j_conf].uniqID,
440  e * env.factor_14lj, sqrt(d2));
441  /* print end */
442 
443  done = 1;
444  break;
445  }
446  }
447  }
448  if (done) continue;
449 
450  /* print large atom-atom vdw */
451  printf("VDW btw \"%s %s\" and \"%s %s\": %8.3f, dist=%8.4f\n",
452  iatom_p->name, prot.res[i_res].conf[i_conf].uniqID,
453  jatom_p->name, prot.res[j_res].conf[j_conf].uniqID,
454  e, sqrt(d2));
455  /* print end */
456 
457  pair_vdw += e;
458  }
459  }
460  }
461 
462  if (i_res == j_res && i_conf == j_conf) return pair_vdw/2.;
463  else return pair_vdw;

466 float coulomb_conf_fast(int i_res, int i_conf, int j_res, int j_conf, PROT prot) {
467  /* This is a fast version of vdw_conf, pre-setup is need to use this function and to make calculation fast,
468  to setup, call the setup functions before get into the vdw loop. See example:
469  */
470  float pair_coulomb = 0., d, d2, e;
471  VECTOR v1,v2;
472  int i_atom, j_atom, i_connect, done;
473  ATOM *iatom_p,*jatom_p;
474 
475  if (!prot.res[i_res].conf[i_conf].n_atom) return pair_coulomb;
476  if (!prot.res[j_res].conf[j_conf].n_atom) return pair_coulomb;
477 
478  for (i_atom=0; i_atom<prot.res[i_res].conf[i_conf].n_atom; i_atom++) {
479  iatom_p = &prot.res[i_res].conf[i_conf].atom[i_atom];
480  if (!iatom_p->on) continue;
481  v1 = iatom_p->xyz;
482  for (j_atom=0; j_atom<prot.res[j_res].conf[j_conf].n_atom; j_atom++) {
483  jatom_p = &prot.res[j_res].conf[j_conf].atom[j_atom];
484  if (!jatom_p->on) continue;
485  v2 = jatom_p->xyz;
486 
487  if (iatom_p == jatom_p) continue;
488  d2 = ddvv(v1, v2);
489  done = 0;
490 
491  if (d2 <= prot.res[i_res].r12sq_max) {
492  for (i_connect=0;i_connect<prot.res[i_res].n_connect12[i_conf][i_atom];i_connect++) {
493  if (jatom_p == prot.res[i_res].connect12[i_conf][i_atom][i_connect]) {
494  done = 1;
495  }
496  }
497  }
498  if (done) continue;
499 
500  if (d2 <= prot.res[i_res].r13sq_max) {
501  for (i_connect=0;i_connect<prot.res[i_res].n_connect13[i_conf][i_atom];i_connect++) {
502  if (jatom_p == prot.res[i_res].connect13[i_conf][i_atom][i_connect]) {
503  done = 1;
504  }
505  }
506  }
507  if (done) continue;
508 
509  /* calculate coulumb */
510  d = sqrt(d2);
511  if (d < 0.8) d = 0.8;
512  e = 331.5*iatom_p->crg*jatom_p->crg/(env.epsilon_coulomb * d);
513 
514  if (d2 <= prot.res[i_res].r14sq_max) {
515  for (i_connect=0;i_connect<prot.res[i_res].n_connect14[i_conf][i_atom];i_connect++) {
516  if (jatom_p == prot.res[i_res].connect14[i_conf][i_atom][i_connect]) {
517 
518  pair_coulomb += e * env.factor_14lj;
519  done = 1;
520  break;
521  }
522  }
523  }
524  if (done) continue;
525 
526  pair_coulomb += e;
527  }
528  }
529 
530  if (i_res == j_res && i_conf == j_conf) return pair_coulomb/2.;
531  else return pair_coulomb;
532 }

551  smaller_max = i_max;

557  if (dist.x < 0) dist.x = 0;
558  if (dist.y < 0) dist.y = 0;
559  if (dist.z < 0) dist.z = 0;
