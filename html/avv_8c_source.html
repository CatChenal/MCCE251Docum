39 double avv(VECTOR v1, VECTOR v2)
40 { double t;
41  v1 = vector_normalize(v1); v2 = vector_normalize(v2);
42  t = v1.x*v2.x + v1.y*v2.y + v1.z*v2.z;
43  if (t>1.0) t = 1.0;
44  else if (t < -1.0) t = -1.0;
45  return acos(t);
46 }
