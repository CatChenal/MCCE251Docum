1 /*
2  * NAME
3  * get_files - list of file names in a directory
4  *
5  * SYNOPSIS
6  * #include <mcce.h>
7  *
8  * STRINGS get_files(char *dir);
9  *
10  * DESCRIPTION
11  * The get_files() function returns a STRINGS object with total number of
12  * files, and a list of file names.
13  *
14  * The STRINGS structure is declared as follows:
15  * typedef struct {
16  * int n; - total number of strings
17  * char **strings; - the list of strings
18  * } STRINGS;
19  *
20  * You should be aware that the field 'strings' is dynamically allocated,
21  * and it is your responsibily to deallocate if necessary. The function
22  * free_strings() is designed to free a STRING object, and reset n to 0.
23  *
24  * This version of get_files() has been tested on Mac OS X and Linux.
25  * The version for irix is get_files_irix(), which is different in:
26  * 1) it includes different header files.
27  * 2) it determines whether a filename is a directory based on *stat,
28  * rather than a *dirent structure.
29  *
30  * RETURN VALUE
31  * The get_files() function returns a STRINGS variable.
32  *
33  * SEE ALSO
34  * free_strings
35  *
36  * EXAMPLE
37  * #include <stdio.h>
38  * #include <mcce.h>
39  * ...
40  * char *mydir = "funalgo";
41  * int i;
42  * STRINGS files;
43  *
44  * files = get_files(mydir);
45  * for (i = 0; i < files.n; i++) {
46  * printf("%d, file : %s\n", i, files.strings[i]);
47  * }
48  * free_strings(files);
49  *
50  * AUTHOR
51  * Yanjun Wang, 06/09/2003
52  */

64 STRINGS get_files(char *dir_name)
65 {
66  STRINGS file_names; /* the value to be returned */
67  DIR *dir; /* DIR returned by opendir() */
68  struct dirent *dir_entry; /* returned by readdir(dir) */
69  FILE *temp; /* a stream to temporarily hold filenames */
70  char s[256]; /* a buffer to hold one filename */
71  int i;
72 
73  temp = tmpfile();
74  file_names.n = 0;
75 
76  /*
77  * read the directory, excluding current directory (.), parent directory (..),
78  * and sub-directories.
79  *
80  * if failed reading the directory:
81  * return to the caller immediately
82  * else
83  * save number of files in 'file_names.n'
84  * save all filenames to 'temp'
85  * malloc space to hold file_names.strings
86  */
87  if ((dir = opendir(dir_name)) == NULL) {
88  printf(" FATAL: get_files(): \"failed reading directory %s.\"\n", dir_name);
89  file_names.strings = NULL;
90  }
91  else {
92  while ((dir_entry = readdir(dir)) != NULL) {
93  if ( strcmp(dir_entry->d_name, ".") != 0 &&
94  strcmp(dir_entry->d_name, "..") != 0 &&
95  dir_entry->d_type != DT_DIR) {
96  file_names.n ++;
97  fprintf(temp, "%s\n", dir_entry->d_name);
98  }
99  }
100  rewind(temp);
101 
102  /* request physical memory to put filenames */
103  file_names.strings = (char **) malloc (file_names.n * sizeof (char *));
104 
105  /* copy filenames from the stream to the STRINGS structure */
106  for (i = 0; i < file_names.n; i++) {
107  fgets(s, sizeof(s), temp);
108  *strchr(s, '\n') = '\0';
109  file_names.strings[i] = (char *) malloc ((1 + strlen(s)) * sizeof (char));
110  strcpy(file_names.strings[i], s);
111  }
112 
113  fclose(temp); /* close the stream */
114  closedir(dir); /* close the directory */
115  }
116 
117  return file_names;
118 }
