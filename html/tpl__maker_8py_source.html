545  with open(filename) as g09_in:

547  if found_charges:
548  if '------------' in line:

551  if m != None:
552  charges.append(m.groups())
553  elif len(charges) > 0:
554  print "Expected to match line: " + line
555  elif 'Fitting point charges to electrostatic potential' in line:
556  found_charges=1
