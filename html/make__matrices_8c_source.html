461  else conf1_type = 'D';

464  if (ematrix->conf[j].uniqID[3] == '+' || ematrix->conf[j].uniqID[3] == '-') conf2_type = 'C'; /* one of "C" or "D" */
465  else conf2_type = 'D';
466 
467  /* in a residue ? */
468  if(strncmp(ematrix->conf[i].uniqID,ematrix->conf[j].uniqID, 3) ||
469  strncmp(ematrix->conf[i].uniqID+5,ematrix->conf[j].uniqID+5, 6)) { /* not in the same residue */
470  if (conf1_type == 'C' && conf2_type == 'C') { /* crg-crg corrected average */
471  /* corrected min */
472  if ((ematrix->pw[i][j].crt * ematrix->pw[i][j].ori < 0 || ematrix->pw[j][i].crt * ematrix->pw[j][i].ori < 0)
473  ||((strchr(ematrix->pw[i][j].mark, '?') && strchr(ematrix->pw[j][i].mark, '?')))) { /* abnormal case, sign flipped or both ? marked */
474  if (fabs(ematrix->pw[i][j].ori)>fabs(ematrix->pw[j][i].ori))
475  ematrix->pw[i][j].ele = ematrix->pw[j][i].ele = ematrix->pw[j][i].ori/1.5;

477  ematrix->pw[i][j].ele = ematrix->pw[j][i].ele = ematrix->pw[i][j].ori/1.5;
478  }

501  }
502  else if (strchr(ematrix->pw[i][j].mark, '?') && !strchr(ematrix->pw[j][i].mark, '?')) {
503  ematrix->pw[i][j].ele = ematrix->pw[j][i].ele = ematrix->pw[j][i].crt;
504  }
505  else if (!strchr(ematrix->pw[i][j].mark, '?') && strchr(ematrix->pw[j][i].mark, '?')) {
506  ematrix->pw[i][j].ele = ematrix->pw[j][i].ele = ematrix->pw[i][j].crt;
507  }
508  else {
509  ematrix->pw[i][j].ele = ematrix->pw[j][i].ele = (ematrix->pw[i][j].crt+ematrix->pw[j][i].crt)/2.0;
510  }
511  }
