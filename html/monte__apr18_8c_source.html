834  env.monte_temp/ROOMT * conflist.conf[kc].e * (eh-conflist.conf[kc].Em) * PH2KCAL/58.0;
835  }

844  + conflist.conf[kc].E_extra
845  + conflist.conf[kc].E_ph

847  + conflist.conf[kc].E_TS; /* Monte Carlo "knows" entropy, we want to "correct" it, so the signe is "+" */

937 void MC(int n)
938 {
939  int cycles, n_total, n_cycle;
940  int i, j, k;
941  register int iters;
942  int mem;
943  int *old_state;
944  float old_E;
945  float dE;
946  float b;
947  int nflips;
948  double H_average;
949 
950  int iflip, ires, iconf; /* iconf is 0 to n of the conf in a res */
951  int old_conf, new_conf; /* old_conf and new_conf are from 0 to n_conf in conflist */
952 
953  b = -KCAL2KT/(env.monte_temp/ROOMT);
954 
955  mem =n_free * sizeof(int);
956  old_state = (int *) malloc(mem);
957  E_minimum = E_state = get_E();
958 
959  /* number of cycles and iters in each cycle */
960  if (env.monte_trace > 0) {
961  cycles = (n-1)/env.monte_trace + 1; /* round up */
962  n_total = cycles*env.monte_trace;
963  n_cycle = env.monte_trace;
964  }
965  else {
966  cycles = 1;
967  n_total = n_cycle = n;
968  }
969  /* clear counters */
970  for (i=0; i<conflist.n_conf; i++) {conflist.conf[i].counter = 0;}
971  H_average = 0.0;
972 
973  for (i=0; i<cycles; i++) {
974  fprintf(fp, "Step %10d, E_minimum = %10.2f, E_running = %10.2f\n",
975  i*n_cycle, E_minimum+E_base, E_state+E_base);
976  fflush(fp);
977  iters = n_cycle;
978  while (iters) {
979  /* save state */
980  old_E = E_state;
981  memcpy(old_state, state, mem);
982 
983  /* 1st flip */
984  ires = rand()/(RAND_MAX/n_free + 1);
985  while (1) {
986  iconf = rand()/(RAND_MAX/free_res[ires].n + 1);
987  old_conf = state[ires];
988  new_conf = free_res[ires].conf[iconf];
989  if (old_conf != new_conf) break;
990  }
991  state[ires] = new_conf;
992  E_state += conflist.conf[new_conf].E_self - conflist.conf[old_conf].E_self;
993  for (j=0; j<n_free; j++) {
994  E_state += pairwise[new_conf][state[j]] - pairwise[old_conf][state[j]];
995  }
996  /* now multiple flip */
997  /* 1st flip -> No (50% probablity)
998  * | Yes (50% probablity)
999  * 2nd flip (any res in big list)
1000  * |
1001  * 3rd flip (any res in big list)
1002  * |
1003  * 4th flip (any res in big list)
1004  */
1005  if (rand() & 1) { /* do multiple flip if odd number */
1006  if (biglist[ires].n) {
1007  nflips = env.monte_flips > (biglist[ires].n+1) ? biglist[ires].n+1: env.monte_flips;
1008  for (k=1; k<nflips; k++) {
1009  iflip = biglist[ires].res[rand()/(RAND_MAX/biglist[ires].n + 1)];
1010  iconf = rand()/(RAND_MAX/free_res[iflip].n + 1);
1011  old_conf = state[iflip];
1012  new_conf = free_res[iflip].conf[iconf];
1013 
1014  state[iflip] = new_conf;
1015  E_state += conflist.conf[new_conf].E_self - conflist.conf[old_conf].E_self;
1016  for (j=0; j<n_free; j++) {
1017  E_state += pairwise[new_conf][state[j]] - pairwise[old_conf][state[j]];
1018  }
1019  }
1020  }
1021  }
1022  if (E_minimum > E_state) { E_minimum = E_state; }
1023 
1024  dE = E_state - old_E;
1025  if (dE < 0.0) { /* go to new low */
1026  }
1027  /*<<< Boltzmann distribution >>>*/
1028  else if ((float) rand()/RAND_MAX < exp(b*dE)) { /* Go */
1029  }
1030  else { /* stay, restore the state */
1031  memcpy(state, old_state, mem);
1032  E_state = old_E;
1033  }
1034  /* count this state energy */
1035  H_average += E_state/n_total;
1036 
1037  /*<<< Do statistics >>>*/
1038  for (j=0; j<n_free; j++) {
1039  conflist.conf[state[j]].counter++;
1040  //+= pairwise_vdw[j]
1041  }
1042  iters --;
1043  }
1044  }
1045  fprintf(fp, "Exit %10d, E_minimum = %10.2f, E_running = %10.2f\n", n_total, E_minimum+E_base, E_state+E_base);
1046  fprintf(fp, "The average running energy, corresponding to H, is %8.3f kCal/mol\n", H_average+E_base);
1047  fflush(fp);
1048 
1049  /* compute occ */
1050  for (i=0; i<conflist.n_conf; i++) {
1051  if (conflist.conf[i].on == 't') continue;
1052  conflist.conf[i].occ = (float) conflist.conf[i].counter / n_total;
1053  }
1054  free(old_state);
1055 
1056  return;
1057 }//void MC(int n)
