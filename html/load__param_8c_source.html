1 /*******************************************************************************
2  * NAME
3  * load_param - load a parameter file into parameter database and convert
4  * the parameter value to appropiate data type.
5  *
6  * SYNOPSIS
7  * #include <mcce.h>
8  *
9  * int load_param(char *fname)
10  *
11  * DESCRIPTION
12  * The load_param() function loads a parameter file to parameter database.
13  * Each entry is a pair of key and value, which can be retrieved later by
14  * the key.
15  *
16  * The key is composed of 3 key strings called key1, key2 and key3. In
17  * general key1 is the record name, key2 is the conformer name, and key3
18  * is the atom name. The leading and ending spaces of 3 keys are stripped
19  * off at the time the parameter entry is saved.
20  *
21  * The value has been either converted to the the appropriate types if
22  * listed below or left as a string and you can process it later. Here is
23  * a table of the converted values:
24  *
25  * record data type
26  * CONNECT CONNECT
27  * CONFLIST STRINGS
28  * IATOM int
29  * PROTON int
30  * ELECTRON int
31  * PKA float
32  * RXN float
33  * CHARGE float
34  * RADIUS float
35  * all other STRING
36  *
37  * A db_open() should be called separately before this function. This is
38  * because multiple parameter files are loaded, but db_open() can be only
39  * called once.
40  *
41  * RETURN VALUE
42  * Return integer 0 on success or -1 on failure.
43  *
44  * SEE ALSO
45  * db_open, db_close, param_sav, param_get, iatom
46  *
47  * EXAMPLE
48  * #include <mcce.h>
49  *
50  * int main()
51  * { db_open();
52  * load_param("asp.txt");
53  * load_param("asn.txt");
54  * db_close();
55  * return 0;
56  * }
57  *
58  * DEPENDENCY
59  * strip_comment, strip_spc, param_sav
60  *
61  * AUTHOR
62  * Junjun Mao, 06/06/2003
63  *******************************************************************************/

85 int load_param(char *fname)
86 { FILE *fp; /* handler of the loaded parameter file */
87  char sbuff[MAXCHAR_LINE]; /* string buffer */
88  char line[MAXCHAR_LINE]; /* line buffer of a parameter line */
89  char *ptr; /* position pointer of the above line */
90  char key1[LEN_KEY1+1]; /* key1 of the parameter entry */
91  char key2[LEN_KEY2+1]; /* key2 of the parameter entry */
92  char key3[LEN_KEY3+1]; /* key3 of the parameter entry */
93  int Counter; /* atom counter */
94  int i;
95 
96  if (check_tpl(fname)) {
97  printf("FATAL: There is an error in tpl file %s\n", fname);
98  return USERERR;
99  }
100 
101  /* open the parameter file, return -1 on failure */
102  if ((fp=fopen(fname, "r")) == NULL) {
103  printf("load_param(): failed oppening file \"%s\".\n", fname);
104  return USERERR;
105  }
106 
107  /* process each valid line in the parameter file */
108  while (fgets(sbuff, MAXCHAR_LINE*sizeof(char), fp)) {
109  /* strip off the comment of the line */
110  memset(line,0,MAXCHAR_LINE*sizeof(char));
111  rm_comment(line, sbuff);
112  if (strncmp(line, "ATOMNAME ",9)) {
113  for(i=strlen(line)-1;i>=0;i--) {
114  if (line[i]==' ') line[i]='\0';
115  else if (line[i]=='\t') line[i]='\0';
116  else break;
117  }
118  }
119 
120  /* if the line is shorter than the total length of 3 keys,
121  * then it has no value, we proceed to the next line */
122  if (strlen(line) < (LEN_KEY1+LEN_KEY2+LEN_KEY3)) continue;
123 
124  /* else we create 3 keys */
125  strncpy(key1,line,LEN_KEY1); key1[LEN_KEY1] = '\0';
126  strncpy(key2,line+LEN_KEY1,LEN_KEY2); key2[LEN_KEY2] = '\0';
127  strncpy(key3,line+LEN_KEY1+LEN_KEY2,LEN_KEY3); key3[LEN_KEY3] = '\0';
128 
129  /* Then we convert the value string to appropiate type */
130  if (!strcmp(key1, "CONNECT ")) { /* convert to CONNECT structure */
131  CONNECT connect; /* temporary CONNECT record */
132  CONNECTED_ATOM connected; /* temporary CONNECTED_ATOM record */
133 
134  /* get the orbital name from column LEN_KEY1+LEN_KEY2+LEN_KEY3, 10 characters */
135  strncpy(sbuff, line+LEN_KEY1+LEN_KEY2+LEN_KEY3, 10); sbuff[10] = '\0';
136  strip(connect.orbital, sbuff);
137 
138  /* set the line pointer to the first atom */
139  if (strlen(line) <= LEN_KEY1+LEN_KEY2+LEN_KEY3 + 10)
140  ptr = line+strlen(line); /* no atom, step to the end */
141  else
142  ptr = line+LEN_KEY1+LEN_KEY2+LEN_KEY3 + 10; /* 1st atom, step to the 1st atom */
143 
144  /* set atom counter */
145  Counter = 0;
146 
147  /* start looping over of atoms */
148  while (strlen(ptr)) {
149  strncpy(sbuff, ptr, 5); sbuff[5] = '\0'; /* cut the first 5 char */
150  connected.res_offset = atoi(sbuff); /* convert to integer */
151  if (strstr(sbuff, "LIG")) connected.ligand = 1;/* ligand flag is on */
152  else connected.ligand = 0; /* ligand flag is off */
153 
154  strncpy(connected.name, ptr+5, 4); /* cut the next 4 char as the atom name */
155  connected.name[4] = '\0';
156  while (strlen(connected.name)<4) /* If atom name is too short, complete with space */
157  strcat(connected.name," ");
158 
159  connect.atom[Counter] = connected; /* save this connected atom */
160  Counter ++; /* Counter steps forward by 1 */
161  if (strlen(ptr) > 10) ptr += 10; /* string pointer steps forward by 1 atom */
162  else ptr += strlen(ptr); /* string pointer steps forward to the end */
163  }
164 
165  /* store the value */
166  connect.n = Counter;
167  param_sav(key1, key2, key3, &connect, sizeof(CONNECT));
168 
169  }
170  else if (!strcmp(key1, "CONFLIST ")) { /* convert to STRINGS */
171  STRINGS value;
172  strip(sbuff, line+LEN_KEY1+LEN_KEY2+LEN_KEY3);
173  value.n = strlen(sbuff)/6+1; /* 5 char for each conformer name, 1 separation space, # of conformers */
174 
175  /* allocate string array from the number of conf names */
176  value.strings = (char **) malloc(value.n*sizeof(char *));
177 
178  /* loop over all conformer names in the value string */
179  for (Counter=0; Counter<value.n; Counter++) {
180  /* allocate the string for 5 char in each conf entry */
181  value.strings[Counter] = (char *) malloc(6*sizeof(char));
182  /* copy the conformer name */
183  strncpy(value.strings[Counter], sbuff+Counter*6, 5);
184  value.strings[Counter][5] = '\0';
185  }
186  /* save this STRINGS structure */
187  param_sav(key1, key2, key3, &value, sizeof(value));
188  }
189  else if (!strcmp(key1, "PROTON ") ||
190  !strcmp(key1, "IATOM ") ||
191  !strcmp(key1, "NATOM ") ||
192  !strcmp(key1, "ELECTRON ")
193  ) { /* if falls into one of them, convert string to float */
194  int value;
195  value = atoi(line+LEN_KEY1+LEN_KEY2+LEN_KEY3);
196  param_sav(key1, key2, key3, &value, sizeof(int));
197  }
198  else if (!strcmp(key1, "PKA ") ||
199  !strcmp(key1, "CHARGE ") ||
200  !strcmp(key1, "RADIUS ") ||
201  !strcmp(key1, "RXN ") ||
202  !strcmp(key1, "VDWAMBER ") ||
203  !strcmp(key1, "VDW_RAD ") ||
204  !strcmp(key1, "VDW_EPS ") ||
205  !strcmp(key1, "RADCOVAL ") ||
206  !strcmp(key1, "EM ") ||
207  !strcmp(key1, "EXTRA ") ||
208  !strcmp(key1, "IPECE_SC ") ||
209  !strcmp(key1, "SCALING ")
210  ) { /* if falls into one of them, convert string to float */
211  float value;
212  value = atof(line+LEN_KEY1+LEN_KEY2+LEN_KEY3);
213  param_sav(key1, key2, key3, &value, sizeof(float));
214  }
215  else if (!strcmp(key1, "CAL_VDW ") ||
216  !strcmp(key1, "RELAX ")
217  ) {
218  int value;
219  if (strchr(line+LEN_KEY1+LEN_KEY2+LEN_KEY3, 't') ||
220  strchr(line+LEN_KEY1+LEN_KEY2+LEN_KEY3, 'T') ) value = 1;
221  else value = 0;
222  param_sav(key1, key2, key3, &value, sizeof(int));
223  }
224  else if (!strcmp(key1, "ROTAMER ") ||
225  !strcmp(key1, "HYDROXYL ")) { /* convert to ROTAMER */
226  ROTAMER value;
227  strncpy(value.atom1, line+LEN_KEY1+LEN_KEY2+LEN_KEY3, 4); value.atom1[4] = '\0';
228  strncpy(value.atom2, line+LEN_KEY1+LEN_KEY2+LEN_KEY3+5, 4); value.atom2[4] = '\0';
229  strip(value.affected, line+LEN_KEY1+LEN_KEY2+LEN_KEY3+10);
230  /* save this STRINGS structure */
231  param_sav(key1, key2, key3, &value, sizeof(value));
232  }
233  else if (!strcmp(key1, "ROT_SWAP ")) {
234  SWAP_RULE value;
235  int i_swap;
236  char *sbuffer;
237  sbuffer = line+LEN_KEY1+LEN_KEY2+LEN_KEY3;
238  value.n_swap = 0;
239  while (strlen(sbuffer) > 5) {
240  value.n_swap++;
241  i_swap = value.n_swap - 1;
242  strncpy(value.swap_atom1[i_swap], sbuffer, 4);
243  value.swap_atom1[i_swap][4] = '\0';
244  strncpy(value.swap_atom2[i_swap], sbuffer+5, 4);
245  value.swap_atom2[i_swap][4] = '\0';
246 
247  if (strlen(sbuffer)>=10) sbuffer+=10;
248  else sbuffer+=(strlen(sbuffer));
249  }
250  param_sav(key1, key2, key3, &value, sizeof(value));
251  }
252  else if (!strcmp(key1, "TORSION ")) {
253  TORS value;
254  char sbuffer[10];
255  int i_term;
256  strncpy(value.atom1, line+LEN_KEY1+LEN_KEY2+LEN_KEY3, 4); value.atom1[4] = '\0';
257  strncpy(value.atom2, line+LEN_KEY1+LEN_KEY2+LEN_KEY3+5, 4); value.atom2[4] = '\0';
258  strncpy(value.atom3, line+LEN_KEY1+LEN_KEY2+LEN_KEY3+10, 4); value.atom3[4] = '\0';
259  strncpy(sbuffer, line+LEN_KEY1+LEN_KEY2+LEN_KEY3+15, 4); sbuffer[4] = '\0';
260  if (strchr(sbuffer, 't') ||
261  strchr(sbuffer, 'T') ) value.opt_hyd = 1;
262  else value.opt_hyd = 0;
263  for(i=strlen(line)-1;i>=0;i--) {
264  if (line[i]==' ') line[i]='\0';
265  else if (line[i]=='\t') line[i]='\0';
266  else break;
267  }
268  value.n_term = ((float)(strlen(line)-(LEN_KEY1+LEN_KEY2+LEN_KEY3+20)))/30.+1;
269  if (value.n_term < 1) value.n_term = 1;
270  for (i_term=0;i_term<value.n_term; i_term++) {
271  strncpy(sbuffer, line+LEN_KEY1+LEN_KEY2+LEN_KEY3+20+i_term*30, 9); sbuffer[9] = '\0'; value.V2[i_term] = atof(sbuffer);
272  strncpy(sbuffer, line+LEN_KEY1+LEN_KEY2+LEN_KEY3+30+i_term*30, 9); sbuffer[9] = '\0'; value.n_fold[i_term] = atof(sbuffer);
273  strncpy(sbuffer, line+LEN_KEY1+LEN_KEY2+LEN_KEY3+40+i_term*30, 9); sbuffer[9] = '\0'; value.gamma[i_term] = env.d2r * atof(sbuffer);
274  }
275  /* save */
276  param_sav(key1, key2, key3, &value, sizeof(TORS));
277  }
278  else { /* others, convert STRING structure */
279  strcpy(sbuff, line+LEN_KEY1+LEN_KEY2+LEN_KEY3);
280  param_sav(key1, key2, key3, sbuff, sizeof(sbuff));
281  /* save with the string terminating character */
282  }
283  }
