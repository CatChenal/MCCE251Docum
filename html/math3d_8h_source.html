147  inline bool SolveQuadratic (scalar_t a, scalar_t b, scalar_t c, scalar_t* x1, scalar_t* x2) {
148  // If a == 0, solve a linear equation
149  if (a == 0) {
150  if (b == 0) return false;
151  *x1 = c / b;
152  *x2 = *x1;
153  return true;
154  } else {
155  scalar_t det = b * b - 4 * a * c;
156  if (det < 0) return false;
157  det = SQRT_FUNCTION(det) / (2 * a);
158  scalar_t prefix = -b / (2*a);
159  *x1 = prefix + det;
160  *x2 = prefix - det;
161  return true;
162  }
163  }

270  inline Matrix4& Matrix4::Rotation (scalar_t angle, Vector4 axis) {
271  scalar_t c = COSINE_FUNCTION(angle);
272  scalar_t s = SINE_FUNCTION(angle);
273  // One minus c (short name for legibility of formulai)
274  scalar_t omc = (1 - c);
275 
276  if (LengthSquared3(axis) != 1) axis = Normalize3(axis);
277 
278  scalar_t x = axis[0];
279  scalar_t y = axis[1];
280  scalar_t z = axis[2];
281  scalar_t xs = x * s;
282  scalar_t ys = y * s;
283  scalar_t zs = z * s;
284  scalar_t xyomc = x * y * omc;
285  scalar_t xzomc = x * z * omc;
286  scalar_t yzomc = y * z * omc;
287 
288  e[0] = x*x*omc + c;
289  e[1] = xyomc + zs;
290  e[2] = xzomc - ys;
291  e[3] = 0;
292 
293  e[4] = xyomc - zs;
294  e[5] = y*y*omc + c;
295  e[6] = yzomc + xs;
296  e[7] = 0;
297 
298  e[8] = xzomc + ys;
299  e[9] = yzomc - xs;
300  e[10] = z*z*omc + c;
301  e[11] = 0;
302 
303  e[12] = 0;
304  e[13] = 0;
305  e[14] = 0;
306  e[15] = 1;

356  inline Matrix4& Matrix4::ProjectionMatrix (bool perspective, scalar_t left_plane, scalar_t right_plane,
357  scalar_t top_plane, scalar_t bottom_plane,
358  scalar_t near_plane, scalar_t far_plane)
359  {
360  scalar_t A = (right_plane + left_plane) / (right_plane - left_plane);
361  scalar_t B = (top_plane + bottom_plane) / (top_plane - bottom_plane);
362  scalar_t C = (far_plane + near_plane) / (far_plane - near_plane);
363 
364  Identity();
365  if (perspective) {
366  e[0] = 2 * near_plane / (right_plane - left_plane);
367  e[5] = 2 * near_plane / (top_plane - bottom_plane);
368  e[8] = A;
369  e[9] = B;
370  e[10] = C;
371  e[11] = -1;
372  e[14] = 2 * far_plane * near_plane / (far_plane - near_plane);
373  } else {
374  e[0] = 2 / (right_plane - left_plane);
375  e[5] = 2 / (top_plane - bottom_plane);
376  e[10] = -2 / (far_plane - near_plane);
377  e[12] = A;
378  e[13] = B;
379  e[14] = C;
380  }
381  return *this;
382  }
