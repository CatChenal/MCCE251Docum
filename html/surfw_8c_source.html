273 /* used to estimate ASA of a residue/conformer when it is fully exposed */
274 /* for multi-conformation, conf[].on flag is used to control which conformer is used for the boundary condition */
275 float get_sas_res(ATOM atom, RES res)
276 {
277  int j, k, m;
278  int count;
279  int status;
280  VECTOR point;
281  float distance2;
282  float surface_area;
283 
284  count = num_pts;
285 
286  for (m = 0; m < num_pts; m++) {
287  status = TRUE;
288  point.x = point_preset[m][0] * atom.vdw_rad + atom.xyz.x;
289  point.y = point_preset[m][1] * atom.vdw_rad + atom.xyz.y;
290  point.z = point_preset[m][2] * atom.vdw_rad + atom.xyz.z;
291 
292  /* go through all atoms in this residue, and excluding point from total count if overlaps with any atom -cmt. by Yifan */
293  for (j = 0; j < res.n_conf; j++) {
294  if (status == FALSE) break;
295  if (j >= 1) { /* only using backbone (j=0) and the conformer that is turned on -added Yifan */
296  if (!res.conf[j].on) continue;
297  }
298  for (k = 0; k < res.conf[j].n_atom; k++) {
299  if (!res.conf[j].atom[k].on) continue;
300  if (res.conf[j].atom[k].vdw_rad < 1e-4) continue; /* ignore 0 radius atoms -Yifan*/
301  if (status != TRUE) continue;
302  if (!memcmp(&atom, &(res.conf[j].atom[k]), sizeof(ATOM))) continue;
303 
304  distance2 = ddvv(point, res.conf[j].atom[k].xyz);
305  if (distance2 < res.conf[j].atom[k].vdw_rad * res.conf[j].atom[k].vdw_rad) {
306  count--;
307  status = FALSE;
308  }
309  }
310  }
311  }
312 
313  surface_area = area_coeff * atom.vdw_rad * atom.vdw_rad * (float)count;
314  return surface_area;
315 }

355 int surfw_res(PROT prot, int i_res, float probe_rad)
356 {
357  int i_conf, j_conf, i_atom;
358  GRID grid;
359 
360  grid.PROBE_RAD = probe_rad; /* get parameters */
361  grid.grid_interval = grid.PROBE_RAD + ATOM_RAD;
362 
363  set_vdw_rad(prot, grid.PROBE_RAD);
364 
365  /* make the surface for each atom */
366  for (i_conf = 1; i_conf < prot.res[i_res].n_conf; i_conf++) {
367  float solution_sas;
368  /* turn on i_conf, and turn off the rest */
369  for (j_conf = 1; j_conf < prot.res[i_res].n_conf; j_conf++) {
370  if (i_conf==j_conf) prot.res[i_res].conf[j_conf].on = 1;
371  else prot.res[i_res].conf[j_conf].on = 0;
372  }
373 
374  get_pdb_size(prot, &grid);
375  extend_grid(&grid);
376  calc_gsize(&grid);
377  //grid.gindex.x = grid.gsize.x - 1;
378  //grid.gindex.y = grid.gsize.y - 1;
379  //grid.gindex.z = grid.gsize.z - 1;
380  alloc_3d_array(grid.gsize.x, grid.gsize.y, grid.gsize.z, &grid);
381  fill_grid(prot, &grid);
382 
383  prot.res[i_res].conf[i_conf].sas = 0.;
384  for (i_atom = 0; i_atom < prot.res[i_res].conf[i_conf].n_atom; i_atom++) {
385  if (!(prot.res[i_res].conf[i_conf].atom[i_atom].on)) continue;
386  if (prot.res[i_res].conf[i_conf].atom[i_atom].vdw_rad < 1e-4) continue; /* ignore 0 radius atoms -Yifan*/
387  mkacc(&(prot.res[i_res].conf[i_conf].atom[i_atom]), &grid);
388  prot.res[i_res].conf[i_conf].sas += prot.res[i_res].conf[i_conf].atom[i_atom].sas;
389  }
390  free_3d_array(grid.gsize.x, grid.gsize.y, grid.gsize.z, &grid);
391 
392  /* calculate the faction of exposure - added Yifan */
393  solution_sas = 0.;
394  for (i_atom = 0; i_atom < prot.res[i_res].conf[i_conf].n_atom; i_atom++) {
395  if (!(prot.res[i_res].conf[i_conf].atom[i_atom].on)) continue;
396  if (prot.res[i_res].conf[i_conf].atom[i_atom].vdw_rad < 1e-4) continue; /* ignore 0 radius atoms -Yifan*/
397  solution_sas += get_sas_res(prot.res[i_res].conf[i_conf].atom[i_atom], prot.res[i_res]);
398  }
399  prot.res[i_res].conf[i_conf].sas_fraction = prot.res[i_res].conf[i_conf].sas/solution_sas;
400  }
401 
402  /* reset control flags in this residue */
403  if (prot.res[i_res].n_conf > 1) prot.res[i_res].conf[1].on = 1;
404  for (i_conf=2; i_conf<prot.res[i_res].n_conf; i_conf++) prot.res[i_res].conf[i_conf].on = 0;
405 
406  reset_atom_rad(prot, grid.PROBE_RAD);
407 
408  return 0;
409

691 /* calculation ASA of terminal N and O atoms of ionizable residues - added Yifan*/
692 int sas_ionizable(PROT prot, float probe_rad)
693 {
694  int i_res, i_conf, i_atom;
695  float score;
696  GRID grid;
697  grid.PROBE_RAD = probe_rad; /* get parameters */
698  grid.grid_interval = grid.PROBE_RAD + ATOM_RAD;
699 
700  delete_h(prot);
701  set_vdw_rad(prot, grid.PROBE_RAD);
702 
703  /* initialize conf[].on */
704  for (i_res=0; i_res<prot.n_res; i_res++) {
705  prot.res[i_res].conf[0].on = 1;
706  if (prot.res[i_res].n_conf > 1) prot.res[i_res].conf[1].on = 1;
707  for (i_conf=2; i_conf<prot.res[i_res].n_conf; i_conf++) prot.res[i_res].conf[i_conf].on = 0;
708  for (i_conf=1; i_conf<prot.res[i_res].n_conf; i_conf++) {
709  prot.res[i_res].conf[i_conf].sas = 0.;
710  prot.res[i_res].conf[i_conf].sas_fraction = 0.;
711  }
712  }
713 
714  /* calculate ASA for all residues */
715  for (i_res=0; i_res<prot.n_res; i_res++) {
716  for (i_conf=1; i_conf<prot.res[i_res].n_conf; i_conf++) {
717  int j_conf;
718  float solution_sas;
719  int use_ipece; /* a flag to decide whether ipece information is used to determine which atom counts as polar */
720  if (i_conf == 1) { /* if the first conformer, check to see if this is an ionizable residue */
721  for (i_atom = 0; i_atom < prot.res[i_res].conf[i_conf].n_atom; i_atom++) {
722  ATOM* atom_p = &prot.res[i_res].conf[i_conf].atom[i_atom];
723  if (!(atom_p->on)) continue;
724  if ( !param_get("IPECE_SC",prot.res[i_res].resName, atom_p->name, &score) ) break;
725  }
726  if (i_atom >= prot.res[i_res].conf[i_conf].n_atom) {
727  /* cannot find an atom matches parameter, move to the next residue */
728  use_ipece = 0;
729  }
730  else {
731  use_ipece = 1;
732  }
733  }
734  /* ionizable residue */
735 
736  /* turn on i_conf, and turn off the rest */
737  for (j_conf = 1; j_conf < prot.res[i_res].n_conf; j_conf++) {
738  if (i_conf==j_conf) prot.res[i_res].conf[j_conf].on = 1;
739  else prot.res[i_res].conf[j_conf].on = 0;
740  }
741 
742  /* make the surface for each atom */
743  get_pdb_size(prot, &grid);
744  extend_grid(&grid);
745  calc_gsize(&grid);
746  alloc_3d_array(grid.gsize.x, grid.gsize.y, grid.gsize.z, &grid);
747  fill_grid(prot, &grid);
748 
749  prot.res[i_res].conf[i_conf].sas = 0.;
750  solution_sas = 0.;
751  for (i_atom = 0; i_atom < prot.res[i_res].conf[i_conf].n_atom; i_atom++) {
752  ATOM* atom_p = &prot.res[i_res].conf[i_conf].atom[i_atom];
753  if (!(atom_p->on)) continue;
754  if (use_ipece) {
755  if ( param_get("IPECE_SC",prot.res[i_res].resName, atom_p->name, &score) != 0) continue; /* skip atoms not in parameter file */
756  }
757  else {
