960

962  dy = y_max - y_min;
963  dz = z_max - z_min;
964 
965  dm = (dx > dy) ? dx : dy;
966  dm = (dm > dz) ? dm : dz;
967  dm += env.radius_probe * 2.0 + 5.0; /* probe and atom radius, assuming no atom is larger than r=2.5 - Yifan */
968
